  def returnInfo
    case @type
    when 'Airy 1830' || '1830'
      a = 6377563.396 
      b = 6356256.910
    when 'Airy 1830 modified' || '1890m'
      a = 6377240.189
      b = 6356034.477
    when 'International 1924' || 'Hayford 1909' || 'i1924'
      a = 6378388.000
      b = 6356911.946
    when 'WGS84' || 'GRS80'
      a = 6378137.000
      b = 6356752.3141
    else
      raise ArgumentError, "Bad input argument", caller
    end
    e_2 = (a**2 - b**2)/a**2
    [a, b, e_2]
  end
