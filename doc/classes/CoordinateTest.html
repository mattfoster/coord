  def test_conversion_module
    # Test the conversions between various coord types.
    # Ignore rounding error.
    test = ConversionTest.new
    val = false

    @types.each {|k, v|
      assert(test.types.include?(k))
      @types.each{|k2, v2|
        if v.class == Array
          v = v.join(', ')
        end
        ret = eval "test.#{k}_to_#{k2}(*[#{v}])"
        if ret.class == Array
          ret.each_with_index {|s, i|
            within_err(s, v2[i])
          }
        else
          within_err(ret, v2)
        end
      }
    }
  end
